Jribbble

A jQuery plugin to fetch data from the Dribbble API - http://dribbble.com/api

Download: Jribbble 0.9.1 Jribbble 0.9.1 + demos

Get a Shot by its Id

$.jribbble.getShotById(playerId, callback)


$.jribbble.getShotById(21188, 
    function (shotDetails) 
    {
        $('#shotById').val(JSON.stringify(shotDetails));
    }
);

Get a List of Shots by the List Name

$.jribbble.getShotsByList(listName, callback, [pagingOpts])


$.jribbble.getShotsByList('popular',
    function (listDetails)
    {
        $('#shotsByList').val(JSON.stringify(listDetails));
    },
    {page: 1, per_page: 5}
);

Get a List of Shots by a Player Id

$.jribbble.getShotsByPlayerId(playerId, callback, [pagingOpts])


$.jribbble.getShotsByPlayerId('tylergaw',
    function (shots)
    {
        $('#shotsByPlayer').val(JSON.stringify(shots));
    },
    {page: 1, per_page: 5}
);
